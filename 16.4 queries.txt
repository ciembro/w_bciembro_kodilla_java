SELECT * FROM ISSUES
JOIN ISSUESLISTS
WHERE ISSUES.ISSUESLIST_ID = ISSUESLISTS.ID;

SELECT ISSUES.ID, ISSUESLIST_ID, SUMMARY, DESCRIPTION, USER_ID_ASSIGNEDTO, USERS.FIRSTNAME, USERS.LASTNAME FROM ISSUES
JOIN USERS 
WHERE ISSUES.USER_ID_ASSIGNEDTO = USERS.ID;

SELECT USERS.FIRSTNAME, USERS.LASTNAME, COUNT(*) AS TASKS FROM ISSUES
JOIN USERS
WHERE USERS.ID = ISSUES.USER_ID_ASSIGNEDTO
GROUP BY USERS.ID;
